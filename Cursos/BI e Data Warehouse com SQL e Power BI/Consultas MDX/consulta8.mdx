(
	{[Produto].[Marca].&[Clean], [Produto].[Marca].&[Linha Refrescante], [Produto].[Marca].&[Sabor da Montanha]}),
	[Measures].[Faturamento], DESC
) ON ROWS,

(
	{[Tempo].[Ano].&[2014]}
) ON COLUMNS

FROM [COMPLETA] WHERE [Measures].[Faturamento]

	{[Tempo].[Ano].&[2014]}
